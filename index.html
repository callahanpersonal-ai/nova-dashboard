"techcrunch.com","theverge.com","arstechnica.com","wired.com" ], keywords: [] } }; // ── BREAK SETTINGS ────────────────────────────────────────────── const BREAK_INTERVAL_MIN = 90; // Alert after 90 min continuous work const IDLE_THRESHOLD_SEC = 300; // 5 min idle = not working const SYNC_INTERVAL_MIN = 30; // Push to Google Sheets every 30 min const DAILY_RESET_HOUR = 4; // Reset daily stats at 4 AM // ── STATE ─────────────────────────────────────────────────────── let sessionStart = Date.now(); let lastActiveTime = Date.now(); let currentCategory = "other"; let isIdle = false; // ── INIT ──────────────────────────────────────────────────────── chrome.runtime.onInstalled.addListener(() => { chrome.storage.local.set({ today: getDateStr(), tracking: {}, // { "work": seconds, "social": seconds, ... } sessions: [], // [{ start, end, category, domain }] dailyLog: [], // [{ date, work, social, entertainment, ... }] totalWorkSession: 0, lastBreakTime: Date.now(), gsheetUrl: "", // User sets this in popup isEnabled: true }); // Set up periodic alarm for syncing + break checks chrome.alarms.create("syncData", { periodInMinutes: SYNC_INTERVAL_MIN }); chrome.alarms.create("breakCheck", { periodInMinutes: 1 }); chrome.alarms.create("dailyReset", { periodInMinutes: 60 }); // Track every 10 seconds for accurate time counting chrome.alarms.create("trackTick", { periodInMinutes: 0.167 }); // ~10 seconds }); // ── CATEGORIZE URL ────────────────────────────────────────────── function categorize(url) { if (!url) return "other"; try { const hostname = new URL(url).hostname.replace("www.", ""); for (const [cat, rules] of Object.entries(CATEGORIES)) { if (rules.domains.some(d => hostname.includes(d))) return cat; if (rules.keywords.some(k => url.toLowerCase().includes(k))) return cat; } } catch (e) {} return "other"; } function getDateStr() { return new Date().toISOString().split("T")[0]; } // ── TRACK ACTIVE TAB ──────────────────────────────────────────── async function trackActiveTab() { const data = await chrome.storage.local.get(["tracking", "today", "isEnabled", "totalWorkSession", "lastBreakTime"]); if (!data.isEnabled) return; // Daily reset check const today = getDateStr(); let tracking = data.tracking || {}; if (data.today !== today) { // Save yesterday's data to dailyLog before resetting await saveDailyLog(data.today, tracking); tracking = {}; await chrome.storage.local.set({ today, tracking, totalWorkSession: 0, lastBreakTime: Date.now() }); } // Get active tab const tabs = await chrome.tabs.query({ active: true, currentWindow: true }); if (!tabs[0]?.url) return; const cat = categorize(tabs[0].url); const nowMs = Date.now(); const elapsed = Math.round((nowMs - lastActiveTime) / 1000); // Only count if less than idle threshold (meaning user was active) if (elapsed > 0 && elapsed < IDLE_THRESHOLD_SEC) { tracking[cat] = (tracking[cat] || 0) + elapsed; // Track continuous work session let workSession = data.totalWorkSession || 0; if (cat === "work") { workSession += elapsed; } else if (["entertainment", "social"].includes(cat) && elapsed > 60) { // Non-work activity for > 1 min counts as a mini-break workSession = Math.max(0, workSession - elapsed); } await chrome.storage.local.set({ tracking, totalWorkSession: workSession }); // Break alert check const workMinutes = workSession / 60; if (workMinutes >= BREAK_INTERVAL_MIN) { chrome.notifications.create("breakAlert", { type: "basic", iconUrl: "icon128.png", title: "N.O.V.A. — Time for a Break", message: `You've been working for ${Math.round(workMinutes)} minutes straight. Stand up, stretch, walk for 10 minutes. Your focus will thank you.`, priority: 2 }); // Reset work session timer after alert await chrome.storage.local.set({ totalWorkSession: 0, lastBreakTime: Date.now() }); } } lastActiveTime = nowMs; currentCategory = cat; } // ── SAVE DAILY LOG ────────────────────────────────────────────── async function saveDailyLog(dateStr, tracking) { const data = await chrome.storage.local.get(["dailyLog"]); const log = data.dailyLog || []; const entry = { date: dateStr, work: Math.round((tracking.work || 0) / 3600 * 10) / 10, social: Math.round((tracking.social || 0) / 3600 * 10) / 10, entertainment: Math.round((tracking.entertainment || 0) / 3600 * 10) / 10, shopping: Math.round((tracking.shopping || 0) / 3600 * 10) / 10, news: Math.round((tracking.news || 0) / 3600 * 10) / 10, other: Math.round((tracking.other || 0) / 3600 * 10) / 10, total: Math.round(Object.values(tracking).reduce((a, b) => a + b, 0) / 3600 * 10) / 10 }; log.push(entry); // Keep last 90 days const trimmed = log.slice(-90); await chrome.storage.local.set({ dailyLog: trimmed }); // Also push to Google Sheets await pushToSheets(entry); } // ── GOOGLE SHEETS SYNC ────────────────────────────────────────── async function pushToSheets(entry) { const data = await chrome.storage.local.get(["gsheetUrl"]); const url = data.gsheetUrl; if (!url) return; try { // Use Google Apps Script Web App endpoint // User deploys a simple Apps Script that accepts POST data await fetch(url, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ sheet: "laptop_screentime", ...entry }) }); } catch (e) { console.error("N.O.V.A. sync error:", e); } } // ── PERIODIC SYNC (push current day's partial data) ───────────── async function periodicSync() { const data = await chrome.storage.local.get(["tracking", "today", "gsheetUrl"]); if (!data.gsheetUrl || !data.tracking) return; const tracking = data.tracking; const entry = { date: data.today, work: Math.round((tracking.work || 0) / 3600 * 10) / 10, social: Math.round((tracking.social || 0) / 3600 * 10) / 10, entertainment: Math.round((tracking.entertainment || 0) / 3600 * 10) / 10, shopping: Math.round((tracking.shopping || 0) / 3600 * 10) / 10, news: Math.round((tracking.news || 0) / 3600 * 10) / 10, other: Math.round((tracking.other || 0) / 3600 * 10) / 10, total: Math.round(Object.values(tracking).reduce((a, b) => a + b, 0) / 3600 * 10) / 10, partial: true }; await pushToSheets(entry); } // ── IDLE DETECTION ────────────────────────────────────────────── chrome.idle.setDetectionInterval(IDLE_THRESHOLD_SEC); chrome.idle.onStateChanged.addListener((state) => { isIdle = state !== "active"; if (state === "active") { lastActiveTime = Date.now(); } }); // ── TAB CHANGE LISTENERS ──────────────────────────────────────── chrome.tabs.onActivated.addListener(() => trackActiveTab()); chrome.tabs.onUpdated.addListener((tabId, changeInfo) => { if (changeInfo.url) trackActiveTab(); }); chrome.windows.onFocusChanged.addListener(() => trackActiveTab()); // ── ALARMS ────────────────────────────────────────────────────── chrome.alarms.onAlarm.addListener((alarm) => { if (alarm.name === "breakCheck") trackActiveTab(); if (alarm.name === "trackTick") trackActiveTab(); if (alarm.name === "syncData") periodicSync(); if (alarm.name === "dailyReset") trackActiveTab(); // triggers daily reset check }); // ── MESSAGE HANDLER (for popup) ───────────────────────────────── chrome.runtime.onMessage.addListener((msg, sender, sendResponse) => { if (msg.type === "getStats") { chrome.storage.local.get(["tracking", "today", "totalWorkSession", "lastBreakTime", "dailyLog"], (data) => { const tracking = data.tracking || {}; const workSessionMin = Math.round((data.totalWorkSession || 0) / 60); const stats = { today: data.today, workHrs: ((tracking.work || 0) / 3600).toFixed(1), socialHrs: ((tracking.social || 0) / 3600).toFixed(1), entertainHrs: ((tracking.entertainment || 0) / 3600).toFixed(1), otherHrs: (((tracking.shopping || 0) + (tracking.news || 0) + (tracking.other || 0)) / 3600).toFixed(1), totalHrs: (Object.values(tracking).reduce((a, b) => a + b, 0) / 3600).toFixed(1), workSessionMin, breakNeeded: workSessionMin >= BREAK_INTERVAL_MIN, dailyLog: data.dailyLog || [] }; sendResponse(stats); }); return true; // async response } if (msg.type === "setGSheetUrl") { chrome.storage.local.set({ gsheetUrl: msg.url }); sendResponse({ ok: true }); } if (msg.type === "toggleEnabled") { chrome.storage.local.get(["isEnabled"], (data) => { chrome.storage.local.set({ isEnabled: !data.isEnabled }); sendResponse({ isEnabled: !data.isEnabled }); }); return true; } if (msg.type === "resetWorkSession") { chrome.storage.local.set({ totalWorkSession: 0, lastBreakTime: Date.now() }); sendResponse({ ok: true }); } });
